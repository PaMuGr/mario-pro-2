154void Demon::update_fireballs(pro2::Window& window, const std::set<const Platform*>& platforms) {
155 for (auto it = fireballs_.begin(); it != fireballs_.end();) {
156 (*it).update(window,platforms);
157 bool erase = false;
158
159 if ((*it).is_expired()) {
160 erase = true;
161 }
162 //Si fora de lloc desactivar
163 else if ((*it).pos().y > window.camera_rect().bottom) {
164 erase = true;
165 }
166 else if ((*it).is_falling()) {
167 for (const Platform* platform : platforms) {
168 pro2::Rect fireballRect = (*it).get_rect();
169 if (interseccionen(fireballRect, platform->get_rect())) {
170 if (randomizer(4, 0) < 3) {
171 (*it).start_fire();
172 }
173 else {
174 erase = true;
175 }
176 break;
177 }
178 }
179 }
180
181 if(erase) it = fireballs_.erase(it);
182 else ++it;
183
184 }
