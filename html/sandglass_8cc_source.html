87 frame_counter_++;

89 if (frame_counter_ % animation_speed_ == 0) {

103 //Quan cooldown = 0 ->> pos random
104 if (is_in_cooldown_) {

107 is_in_cooldown_ = false;
108
109 pro2::Rect camera = window.camera_rect();
110 //Agafem els punts del rectangle de camera
111 int cam_left = camera.left;
112 int cam_right = camera.right;
113 int cam_top = camera.top;
114 int cam_bottom = camera.bottom;
115
116 //Calculem un marge de 20% aixi el objecte estarà a 10% de dalt i baix i dels costats
117 int width_margin = (cam_right - cam_left) * 0.1;
118 int height_margin = (cam_bottom - cam_top) * 0.1;
119
120 //Randomitzem els valors
121 int new_x = randomizer(cam_right - width_margin, cam_left + width_margin);
122
123 //Ens assegurem que no estigui massa abaix
124 int min_y = cam_top + height_margin*2 + 50;
125 int max_y = cam_top + (cam_bottom - cam_top) * 3 / 5;
126 int new_y = randomizer(max_y, min_y);
127
128 //Actualitzem posicions
129 pos_.x = new_x;
130 pos_.y = new_y;
131 }
