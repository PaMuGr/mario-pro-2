46void Game::generate_level_objects(int count) {
47 for (int i = 1; i < count; i++) {
48 int cy = randomizer(-3, 3);
49 int random_height = randomizer(-3, 3);
50 int y_offset = random_height * 10;
51 int base_x = 250 + i * 200;
52 int x_range = 400 + i * 200;
53
54 // Plataforma d'abaix
55 platforms_.push_back(Platform(
56 base_x, x_range,
57 150 + y_offset, 161 + y_offset
58 ));
59
60 int platforms_offset_x = 0;
61 int platforms_offset_y = 0;
62 if (randomizer(1, 0) == 0) {
63 platforms_offset_x = 30;
64 platforms_offset_y = 10;
65 } else {
66 platforms_offset_x = -30;
67 platforms_offset_y = 0;
68 }
69
70 // Plataforma de dalt
71 platforms_.push_back(Platform(
72 base_x + platforms_offset_x, x_range + platforms_offset_x - 20,
73 90 + y_offset - platforms_offset_y, 101 + y_offset - platforms_offset_y
74 ));
75
76 // Creu aleatòria
77 if (randomizer(1, 0) == 0) {
78 crosses_.push_back(Cross({
79 (randomizer(380, 100) + i * 200 + x_range) / 2,
80 120 + random_height * 10
81 }, {1, 0}, 20));
82 } else {
83 crosses_.push_back(Cross({
84 (randomizer(380, 100) + i * 200 + x_range) / 2,
85 50 + random_height * 10
86 }, {((i % 2) * -1 + 1), ((i % 2) * -1)}, 10));
87 }
88 }

92void Game::process_keys(pro2::Window& window) {
93 if (window.is_key_down(Keys::Escape)) {
94 game_state_ = 4; // exit
95 return;
96 }
97 if (game_state_ == 1 && window.was_key_pressed('P')) {
98 game_state_ = 6; // pause
99 return;
100 }
101 if ((game_state_ == 2 || game_state_ == 3 || game_state_ == 5) && window.is_key_down('R')) {
102 int prev_mode = game_mode_;
103 *this = Game(480, 320);
104 game_mode_ = prev_mode;
105 apply_game_mode_settings();
106 game_state_ = 0;
107 return;
108 }
109 if (window.was_key_pressed('B') && ascended_.can_activate()) {
110 ascended_.activate({mario_.pos().x, mario_.pos().y});
111 }
112 if (window.was_key_pressed(Keys::Return)) {
113 game_state_ = 1;
114 return;
115 }
116}

118void Game::apply_game_mode_settings() {
119 switch (game_mode_) {

142 timer_ = 0;

145 break;
146 default:
147 game_state_ = 4;

244/*UPDATE DE LA CAMERA*/
245void Game::update_camera(pro2::Window& window) {
246 const Pt pos = mario_.pos();
247 const Pt cam = window.camera_center();
248
249 const int left = cam.x - window.width() / 4;
250 const int right = cam.x + window.width() / 4;
251 const int top = cam.y - window.height() / 4;
252 const int bottom = cam.y + window.height() / 4;
253
254 int dx = 0, dy = 0;
255 if (pos.x > right) {
256 dx = pos.x - right;
257 } else if (pos.x < left) {
258 dx = pos.x - left;
259 }
260 if (pos.y < top) {
261 dy = pos.y - top;
262 } else if (pos.y > bottom) {
263 dy = pos.y - bottom;
264 }
265
266 window.move_camera({dx, dy});
267}
