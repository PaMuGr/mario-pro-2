107void Game::update_camera(pro2::Window& window) {
108 const Pt pos = mario_.pos();
109 const Pt cam = window.camera_center();
110
111 const int left = cam.x - window.width() / 4;
112 const int right = cam.x + window.width() / 4;
113 const int top = cam.y - window.height() / 4;
114 const int bottom = cam.y + window.height() / 4;
115
116 int dx = 0, dy = 0;
117 if (pos.x > right) {
118 dx = pos.x - right;
119 } else if (pos.x < left) {
120 dx = pos.x - left;
121 }
122 if (pos.y < top) {
123 dy = pos.y - top;
124 } else if (pos.y > bottom) {
125 dy = pos.y - bottom;
126 }
127
128 window.move_camera({dx, dy});
